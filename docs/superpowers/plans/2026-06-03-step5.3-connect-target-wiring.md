```go
package main

import (
	"errors"
	"strings"
)

// valueFlags are the connect/setup flags that consume the FOLLOWING arg as a
// value, so hostFromArgs does not mistake that value for the host.
var valueFlags = map[string]bool{"--port": true, "--local-bin": true}

// errNoHost is returned when no positional host token is present.
var errNoHost = errors.New("missing <host>: usage: cc-clip <connect|setup> <host> [flags]")

// hostFromArgs returns the first non-flag positional token in args (which are
// os.Args[2:]), skipping --flag, --flag=value, and the value after a value-flag.
// This makes flag-before-host (e.g. `connect --codex myhost`) work, replacing
// the positional os.Args[2] assumption.
func hostFromArgs(args []string) (string, error) {
	for i := 0; i < len(args); i++ {
		a := args[i]
		if strings.HasPrefix(a, "--") {
			name := a
			if eq := strings.IndexByte(a, '='); eq >= 0 {
				name = a[:eq]
			} else if valueFlags[a] && i+1 < len(args) {
				i++ // consume the value token
			}
			_ = name
			continue
		}
		if strings.HasPrefix(a, "-") { // short flags (e.g. -v) — not a host
			continue
		}
		return a, nil
	}
	return "", errNoHost
}
```
